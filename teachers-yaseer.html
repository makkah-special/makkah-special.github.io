// Service worker cleanup for GitHub Pages. // This file intentionally clears old caches and unregisters the previous service worker. self.addEventListener('install', (event) => { self.skipWaiting(); }); self.addEventListener('activate', (event) => { event.waitUntil( caches.keys() .then((keys) => Promise.all(keys.map((key) => caches.delete(key)))) .then(() => self.registration.unregister()) .then(() => self.clients.matchAll({ type: 'window', includeUncontrolled: true })) .then((clients) => { clients.forEach((client) => { if (client.url) client.navigate(client.url); }); }) ); });
